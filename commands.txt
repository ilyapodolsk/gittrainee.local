git init - создать репозиторий
git clone <ссылка> - клонировать удаленный репозиторий к себе
git status - проверить какие файлы изменены
git add . - добавить все файлы в индекс (можно по одному, напри git add error.php)
git commit -m "<комментарий>" - сохранить изменения
git remote add origin <ссылка> - подключиться к удаленному репозиторию (достаточно сделать один раз)
git push origin master - отправить все свои коммиты в удаленный репозиторий
